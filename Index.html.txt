<script>
const BRAIN_URL = "YOUR_RAILWAY_URL_HERE"; // ← paste your Railway URL
const WAKE_WORD = "muppuri";

let recognition, listening = false, synth = window.speechSynthesis;

function speak(text){
  synth.cancel();
  let u = new SpeechSynthesisUtterance(text);
  u.rate = 1; u.pitch = 1; synth.speak(u);
}

function setStatus(t, col="#aaa"){
  document.getElementById("status").style.color = col;
  document.getElementById("status").textContent = t;
}

async function sendCommand(cmd){
  setStatus("MUPPURI is thinking...", "#0066ff");
  document.getElementById("orb-text").textContent = "THINKING";
  const secret = document.getElementById("secret").value;
  try {
    const r = await fetch(BRAIN_URL+"/voice", {
      method:"POST", headers:{"Content-Type":"application/json"},
      body: JSON.stringify({transcript: cmd, secret: secret})
    });
    const d = await r.json();
    if(d.error){ setStatus("Wrong secret word","#ff4444"); return; }
    const reply = d.reply;
    document.getElementById("reply").style.display="block";
    document.getElementById("reply").textContent = reply;
    speak(reply);
    setStatus("MUPPURI replied", "#00ff88");
    document.getElementById("orb-text").textContent = "LISTENING";
    // add to history
    const h = document.createElement("div");
    h.className="hist-item";
    h.innerHTML=`<div class="hist-cmd">YOU: ${cmd}</div>MUPPURI: ${reply.substring(0,120)}...`;
    document.getElementById("hist").prepend(h);
  } catch(e){ setStatus("Connection error — check Railway URL","#ff4444"); }
}

function startRecognition(){
  const SR = window.SpeechRecognition || window.webkitSpeechRecognition;
  if(!SR){ alert("Use Chrome browser for voice recognition"); return; }
  recognition = new SR();
  recognition.continuous = true;
  recognition.interimResults = false;
  recognition.lang = "en-IN";
  recognition.onresult = (e) => {
    const t = e.results[e.results.length-1][0].transcript.toLowerCase().trim();
    setStatus("Heard: "+t, "#ffaa00");
    if(t.includes(WAKE_WORD)){
      const cmd = t.split(WAKE_WORD).slice(1).join(" ").trim();
      if(cmd.length > 2){ speak("Yes, processing your command."); sendCommand(cmd); }
      else { setStatus("Say MUPPURI followed by your command","#ffaa00"); speak("Yes? What do you need?"); }
    }
  };
  recognition.onerror = () => { setStatus("Mic error — tap to retry","#ff4444"); listening=false; };
  recognition.onend = () => { if(listening) recognition.start(); };
  recognition.start();
}

function toggleListen(){
  listening = !listening;
  const orb = document.getElementById("orb");
  if(listening){
    orb.classList.add("listening");
    document.getElementById("orb-text").textContent="LISTENING";
    setStatus("Listening... say MUPPURI [command]","#00ff88");
    startRecognition();
  } else {
    orb.classList.remove("listening");
    document.getElementById("orb-text").textContent="TAP TO WAKE";
    setStatus("Tap the orb to activate","#aaa");
    if(recognition) recognition.stop();
    synth.cancel();
  }
}

// Auto-wake on voice even before tap (Chrome only)
document.addEventListener("DOMContentLoaded", ()=>{
  setStatus("Enter secret word above, then tap the orb");
});
